◆動作確認　テーブルの作成
create table users {
id int,
name varchar(255),
email varchar(255),
password char(32)
};

◆テーブルを確認
show tables;

◆テーブルを削除
drop table users;
show tables;


◆テーブルを作成


create table users {
id int not null auto_increment primary key,
name varchar(255),
email varchar(255) unique,
password char(32),
score double,
sex enum('male', 'female') default 'male,
memo text,
created datetime,
key score (score)
};
